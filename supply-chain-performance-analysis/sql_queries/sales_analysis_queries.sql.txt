USE SupplyChainDB;

-- Data Quality Check
SELECT
  SUM(ProcuredDate IS NULL) AS procured_nulls,
  SUM(OrderDate IS NULL) AS order_nulls,
  SUM(ShipDate IS NULL) AS ship_nulls,
  SUM(DeliveryDate IS NULL) AS delivery_nulls
FROM SupplyChainSales;

-- Revenue By Channel
SELECT SalesChannel,
       SUM(OrderQuantity * UnitPrice * (1 - DiscountApplied)) AS Revenue
FROM SupplyChainSales
GROUP BY SalesChannel;

-- Monthly Revenue Trend
SELECT YEAR(OrderDate) AS Year,
       MONTH(OrderDate) AS Month,
       SUM(OrderQuantity * UnitPrice * (1 - DiscountApplied)) AS Revenue
FROM SupplyChainSales
GROUP BY YEAR(OrderDate), MONTH(OrderDate)
ORDER BY Year, Month;

-- Average Delivery Time
SELECT WarehouseCode,
       AVG(DATEDIFF(DeliveryDate, ShipDate)) AS AvgDays
FROM SupplyChainSales
GROUP BY WarehouseCode;

-- Profit Analysis
SELECT OrderNumber,
       (OrderQuantity * UnitPrice * (1 - DiscountApplied))
     - (OrderQuantity * UnitCost) AS Profit
FROM SupplyChainSales;
